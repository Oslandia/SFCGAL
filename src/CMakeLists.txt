FILE(GLOB_RECURSE SFCAL_HEADERS "${CMAKE_SOURCE_DIR}/src/*.h")
FILE(GLOB_RECURSE SFCAL_SOURCES "${CMAKE_SOURCE_DIR}/src/*.cpp")

add_library( SFCGAL
    ${SFCAL_HEADERS}
    ${SFCAL_SOURCES}
)

target_link_libraries( SFCGAL
    ${CGAL_LIBRARIES}
    ${Boost_LIBRARIES}
    ${MPFR_LIBRARIES}
    ${GMP_LIBRARIES}
)

install(TARGETS SFCGAL 
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    BUNDLE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    COMPONENT runtime
)

install(DIRECTORY "${CMAKE_SOURCE_DIR}/src/" DESTINATION include
    COMPONENT development
    FILES_MATCHING PATTERN "*.h"
)
